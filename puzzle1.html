function mystery(arr) {
    let res = [];
    for (let i = 0; i < arr.length; i++) {
        let v = arr[i] * (i % 2 === 0 ? 2 : 3);
        if (v % 4 === 0) res.unshift(v / 2);
        else if (v % 3 === 0) res.push(v - 1);
    }
    return res.filter(x => x % 2 === 0).map(x => x + '').reverse().join('~');
}
